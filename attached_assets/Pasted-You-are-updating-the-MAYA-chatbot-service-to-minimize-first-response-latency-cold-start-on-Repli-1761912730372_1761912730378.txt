Startup warmup (template):

@app.on_event("startup")
def startup():
    app.state.chroma = PersistentClient(path=os.getenv("CHROMA_DIR","/data/chroma"),
                                        settings=Settings(allow_reset=False, anonymized_telemetry=False))
    app.state.col = app.state.chroma.get_or_create_collection(os.getenv("CHROMA_COLLECTION","maya"))

    app.state.embed = SentenceTransformer("sentence-transformers/all-MiniLM-L6-v2")
    _ = app.state.embed.encode(["warmup"])

    app.state.groq = Groq(api_key=os.getenv("GROQ_API_KEY"))
    try:
        app.state.groq.chat.completions.create(
            model="llama-3.3-70b-versatile",
            messages=[{"role":"user","content":"ping"}],
            max_tokens=1,
            temperature=0
        )
    except Exception:
        pass

    try:
        app.state.col.query(query_texts=["hello"], n_results=1)
    except Exception:
        pass

Turn-1 retrieval & streaming (template):

def retrieve(query: str, k:int=3) -> str:
    qv = app.state.embed.encode([query]).tolist()[0]
    res = app.state.col.query(query_embeddings=[qv], n_results=k)
    docs = []
    for i in range(len(res.get("documents",[[]])[0])):
        doc = res["documents"][0][i]
        md  = res.get("metadatas",[[{}]])[0][i]
        docs.append(f"[{md.get('title','doc')}]\n{doc}")
    return "\n\n---\n\n".join(docs)

async def stream_groq(prompt: str):
    with app.state.groq.chat.completions.stream(
        model="llama-3.3-70b-versatile",
        messages=[{"role":"system","content":"You are MAYA, a concise UK autism support assistant."},
                  {"role":"user","content": prompt}],
        temperature=0.2,
        max_tokens=600
    ) as stream:
        for event in stream:
            if event.type == "token":
                yield event.token.encode()
            elif event.type == "error":
                yield f"\n[error] {event.error}\n".encode()

@app.post("/chat")
async def chat(prompt: str):
    context = retrieve(prompt, k=3)
    full_prompt = f"Context:\n{context}\n\nUser question:\n{prompt}\n\nAnswer clearly; cite titles when helpful."
    return StreamingResponse(stream_groq(full_prompt), media_type="text/plain")
